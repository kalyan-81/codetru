void main() {
  
  //ternary operator
  
  int age=20;
  
  bool greater=age>20?true:false;
  
  print(greater);
  
  
  //switch case 
  
  switch(age){
    case 1:
      print(1);
      break;
    case 2:
      print(2);
      break;
    case 20:
      print(20);
      break;
    default:
      print("none of the cases matched");
  
  }
  
}
